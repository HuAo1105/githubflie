## 数组去重

```js
function quchong(array) {
            var arr1 = [];
            for (var i = 0; i < array.length; i++) {
                if (arr1.indexOf(array[i]) == -1) {
                    arr1.push(array[i]);
                }

            }
            return arr1;
        }
        var arr = [1, 2, 2, 4, 9, 6, 7, 5, 2, 3, 5, 6, 5];
        console.log(quchong(arr));
```

## sort排序

```js
降序
var arr = [3, 6, 2, 4, 1, 5];
    var arr1 = arr.sort(
        function(a, b) {
            return b - a;
        }
    )
    console.log(arr1);

升序
var arr = [3, 6, 2, 4, 1, 5];
    var arr1 = arr.sort(
        function(a, b) {
            return a - b;
        }
    )
console.log(arr1);
```

## 查找数组中重复出现

```
var arr = ["c", "a", "z", "a", "x", "a"];
    var o = {};// c:1 a:3 z:1 x:1
    for (var i = 0; i < arr.length; i++) {
        var item = arr[i];
        if (o[item]) {
            o[item] = o[item] + 1;
            // o[item] ++;这个也可以
        } else {
            o[item] = 1;
        }
    }
```
